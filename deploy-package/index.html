AI Agent Integration Example
// Add to your AI agent
import { CreatorPay } from '@creatorpay/sdk';

const client = new CreatorPay({
  apiKey: process.env.CREATORPAY_API_KEY,
  wallet: process.env.AI_AGENT_WALLET
});

// Automatically handle paywalled content
async function accessContent(url) {
  const content = await client.getContent(url);
  
  if (content.paywall) {
    // Auto-pay via x402 protocol
    await client.pay({
      amount: content.price,
      recipient: content.creator
    });
    
    return await client.getContent(url);
  }
  
  return content;
}
